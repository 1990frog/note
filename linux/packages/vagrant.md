# 添加box（镜像）
vagrant box add centos/7

# 初始化
vagrant init centos/7

# 启动
vagrant up

# 停止
vagrant halt

# 销毁
vagrant destroy

# 打开tty
vagrant ssh

# 查看状态
vagrant status

# 查看ssh
vagrant ssh-config
