# 中州韵
配置文件
## 基础文件
```yaml
patch:
    # 候选词
    menu/page_size: 9 #设置候选字数量，根据喜好调整。
    speller/algebra:
    # 模糊音
    # 平翘舌
    - derive/zh&/z/
    - derive/ch&/c/
    - derive/sh&/s/
    
    # 模糊音
    - derive/([ei])n$/$1ng/    # ing = in; eng = en
    - derive/([ei])ng$/$1n/    # in = ing; en = eng

    # 基础
    - abbrev/^([a-z]).+$/$1/
    - abbrev/^([zcs]h).+$/$1/
    
    # 补全
    - derive/([dtngkhrzcs])o(u|ng)$/$1o/   # o = ou; o = ong
    - derive/ong$/on/      # on = ong
    - derive/^ding$/din/     # din = ding
    
    # 处理 v 和 u
    - derive/^([nl])ue$/$1ve/   # nve = nue; lve = lue
    - derive/^([jqxy])u/$1v/    # v = u; v = u
    
    # 智能纠错
    - derive/ao$/oa/       # oa = ao
    - derive/([iu])a(o|ng?)$/a$1$2/   # aio = iao; aing = iang; aung = uang
    - derive/([aeiou])ng$/$1gn/   # gn = ng
    - derive/un$/uen/    # uen = un
    - derive/ui$/uei/    # uei = ui
    - derive/iu$/iou/    # iou = ui
    - derive/tie$/tei/    # tei = tie
    - derive/i$/ii/      # ii = i  # i 不小心按两下
    - derive/u$/uu/      # ui = u  # u 不小心按两下
```

## 词库
```yaml
---
name: luna_pinyin.my_words    # 词库名称，与文件名一致
version: "1.0"                # 版本号，任填
sort: by_weight               # 排序依据，by_weight表示按权重（词频）排序
use_preset_vocabulary: true   # 是否使用预置发音，一般选true
# 此处为扩充词库默认一并载入的词库
import_tables:
  - luna_pinyin               # 预置的朙月拼音词库
  - zhwiki                    # 第三方词库举例1：维基百科词库
  - moegirl                   # 第三方词库举例2：萌娘百科词库
---
```
